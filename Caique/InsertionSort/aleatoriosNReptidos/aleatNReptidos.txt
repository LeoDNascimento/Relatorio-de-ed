bool existe(int valores[], int tam, int valor){
for(int i=0;i<tam;i++){
    if(valores[i]==valor)
        return true;
}
return false;
}

void aleatorios(int Vector[],int quantNumeros, int Limite){
srand(time(NULL));
int v;
for(int i=0;i<quantNumeros;i++){
    v=rand()%Limite;
    while(existe(Vector,i,v)){
        v=rand()%Limite;
                }
    Vector[i]=v;
}

}

void insertSort(int *v, int n, int *info){
    int i, j, aux;
    for(i=1; i<n; i++){
        aux=v[i];
        for(j=i; (j>0) && (aux<v[j-1]); j--)
            v[j]=v[j-1];
             v[j]=aux;
             info[0]++;
             info[1]++;


    }
}

void imprime(int vetor[]){
for(int j=0;j<10;j++){
    printf("\n%d", vetor[j]);
}
printf("\n");
}
