void aleatorios(int Vector[],int quantNumeros, int Limite){
    srand(time(NULL));
    for(int i=0;i<quantNumeros;i++){
        Vector[i] = rand()%Limite;
}

}

void shellSort(int *vet, int size, int *info){
    int i, j, value;
    int gap = 1;
    while(gap < size){
        gap = 3*gap+1;
    }
    while ( gap > 1) {
        gap /= 3;
        for(i = gap; i < size; i++) {
            value = vet[i];
            j = i;
            while (j >= gap && value < vet[j - gap]) {
                vet[j] = vet [j - gap];
                j = j - gap;
                info[0]++;
                info[1]++;
            }
            vet [j] = value;
        }
    }
}

void imprime(int vetor[]){
for(int j=0;j<10;j++){
    printf("\n%d", vetor[j]);
}
printf("\n");
}
