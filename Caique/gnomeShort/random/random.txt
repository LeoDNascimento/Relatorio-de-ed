void swap(int *A, int *B){

* A = *B;
* B = C;
}

void GnomeSort(){
 int next = 0, previous = 0;
 int i = 0;

 for(i = 0; i < 10 ; i++) {
comparacao++;
 if(VectorSort[i] > VectorSort[i + 1]) {
  previous = i;
  next = i + 1;

  while(VectorSort[previous] > VectorSort[next])  {
 swap(&VectorSort[previous],&VectorSort[next]);
 troca++;
 if(previous > 0)
    previous--;
 if(next > 0)
    next--;
  }
 }

 }
