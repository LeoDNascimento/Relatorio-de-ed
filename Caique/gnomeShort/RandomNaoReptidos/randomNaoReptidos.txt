void swap(int *A, int *B){
 int C = *A;
* A = *B;
* B = C;
}

bool existe(int valores[], int tam, int valor){
for(int i=0;i<tam;i++){
    if(valores[i]==valor)
        return true;
}
return false;
}

void aleatorios(int VectorSort[],int quantNumeros, int Limite){
srand(time(NULL));
int v;
for(int i=0;i<quantNumeros;i++){
    v=rand()%Limite;
    while(existe(VectorSort,i,v)){
        v=rand()%Limite;
                }
    VectorSort[i]=v;
}

}

void imprime(){
for(int j=0;j<10;j++){
    printf("\n%d", VectorSort[j]);
}
printf("\n");
}

void GnomeSort(){
 int next = 0, previous = 0;
 int i = 0;

 for(i = 0; i < 10 ; i++) {
comparacao++;
 if(VectorSort[i] > VectorSort[i + 1]) {
  previous = i;
  next = i + 1;

  while(VectorSort[previous] > VectorSort[next])  {
 swap(&VectorSort[previous],&VectorSort[next]);
 troca++;
 if(previous > 0)
    previous--;
 if(next > 0)
    next--;
  }
 }

 }



}
